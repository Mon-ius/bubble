%% preamble.tex — document class, packages, and AER-style layout
%%
%% This file is \input{} from main.tex before any section is loaded.
%% All formatting concerns (margins, fonts, section headings, display
%% math spacing, hyperref, bibliography style) live here, so that the
%% section files contain only content and main.tex stays minimal.

\documentclass[11pt]{article}

\usepackage[top=0.85in,bottom=0.9in,left=0.95in,right=0.95in]{geometry}
\usepackage{times}
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{amsmath,amssymb,amsthm}
\usepackage{graphicx}
\usepackage{booktabs}
\usepackage{array}
\usepackage{multirow}
\usepackage{caption}
\usepackage{subcaption}
\usepackage{setspace}
\usepackage{titlesec}
\usepackage{titling}
\usepackage{natbib}
\usepackage[hidelinks]{hyperref}
\usepackage{url}

% --- Title block spacing ------------------------------------------
\setlength{\droptitle}{-36pt}

% --- AER-style heading conventions --------------------------------
\renewcommand{\thesection}{\Roman{section}}
\renewcommand{\thesubsection}{\Alph{subsection}}
\renewcommand{\thesubsubsection}{\arabic{subsubsection}}
\titleformat{\section}{\normalfont\normalsize\bfseries\centering}{\thesection.}{0.5em}{}
\titleformat{\subsection}{\normalfont\normalsize\bfseries}{\thesubsection.}{0.5em}{}
\titleformat{\paragraph}[runin]{\normalfont\normalsize\itshape}{}{0pt}{}[.]
\titlespacing*{\section}{0pt}{0.8\baselineskip}{0.35\baselineskip}
\titlespacing*{\subsection}{0pt}{0.5\baselineskip}{0.25\baselineskip}
\titlespacing*{\paragraph}{0pt}{0.35\baselineskip}{0.4em}

% --- Paragraph and display-math spacing ---------------------------
\setlength{\parskip}{0pt}
\setlength{\parindent}{18pt}
\linespread{0.97}
\AtBeginDocument{%
  \setlength{\abovedisplayskip}{4pt}%
  \setlength{\belowdisplayskip}{4pt}%
  \setlength{\abovedisplayshortskip}{2pt}%
  \setlength{\belowdisplayshortskip}{2pt}%
}
